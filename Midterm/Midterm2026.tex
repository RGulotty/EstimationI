\begin{enumerate}

%%% PROBLEM 1 %%%
\item Suppose $\bm{P} = \bm{X}(\bm{X}'\bm{X})^{-1}\bm{X}'$ is the projection matrix and $\bm{M} = \bm{I} - \bm{P}$ is the annihilator matrix.

\begin{enumerate}
  \item Suppose $\bm{y}$ is a linear combination of the columns of $\bm{X}$, i.e., $\bm{y} = \bm{X}\bm{c}$ for some vector $\bm{c}$. What are the OLS coefficient estimate $\hat{\bm{\beta}}$, the fitted values $\hat{\bm{y}}$, and the residuals $\hat{\bm{e}}$?

  \textbf{Answer.} $\hat{\bm{\beta}} = (\bm{X}'\bm{X})^{-1}\bm{X}'\bm{X}\bm{c} = \bm{c}$. The fitted values are $\hat{\bm{y}} = \bm{P}\bm{y} = \bm{P}\bm{X}\bm{c} = \bm{X}\bm{c} = \bm{y}$ (since $\bm{PX} = \bm{X}$). The residuals are $\hat{\bm{e}} = \bm{M}\bm{y} = \bm{y} - \hat{\bm{y}} = \bm{0}$. When $\bm{y}$ lies entirely in the column space of $\bm{X}$, the regression fits perfectly.

  \item Now suppose $\bm{z}$ is orthogonal to every column of $\bm{X}$, i.e., $\bm{X}'\bm{z} = \bm{0}$. What are $\bm{Pz}$ and $\bm{Mz}$?

  \textbf{Answer.} $\bm{Pz} = \bm{X}(\bm{X}'\bm{X})^{-1}\bm{X}'\bm{z} = \bm{X}(\bm{X}'\bm{X})^{-1}\bm{0} = \bm{0}$. And $\bm{Mz} = (\bm{I} - \bm{P})\bm{z} = \bm{z} - \bm{0} = \bm{z}$.  A vector orthogonal to the column space of $\bm{X}$ has no component to project onto, so it passes through $\bm{M}$ unchanged.
\end{enumerate}

%%% PROBLEM 2 %%%
\item The \emph{conditional expectation function} (CEF) $m(x) = \mathbb{E}[Y \mid X = x]$ and the \emph{best linear predictor} (BLP) $P(Y \mid X) = \alpha + \beta X$ are both tools for predicting $Y$ from $X$, but they are not the same object.

\begin{enumerate}
  \item Under what condition is the BLP equal to the CEF?

  \textbf{Answer.} The BLP equals the CEF when the CEF is linear in $X$. A sufficient condition is that $(X, Y)$ are jointly normally distributed, in which case $\mathbb{E}[Y \mid X] = \alpha + \beta X$ with $\beta = \text{Cov}(X,Y)/\text{Var}(X)$.

  \item Give a specific example of a data-generating process where $\mathbb{E}[Y \mid X]$ is nonlinear in $X$.

  \textbf{Answer.} $Y = X^2 + \epsilon$ with $X \sim N(0,1)$ and $\epsilon$ independent of $X$.  Then $\mathbb{E}[Y \mid X] = X^2$, which is nonlinear.  In fact, the BLP slope is $\beta = \text{Cov}(X, X^2)/\text{Var}(X) = \mathbb{E}[X^3]/1 = 0$ by the symmetry of the normal distribution, so the BLP is just the constant $\mathbb{E}[Y] = 1$.  The linear predictor completely misses the U-shaped relationship.
\end{enumerate}

%%% PROBLEM 3 %%%
\item A researcher runs a randomized experiment (treatment $d_i$ assigned independently of all covariates) and considers two regressions:

\begin{align*}
  \text{Short:} \quad y_i &= \gamma_0 + \gamma_1 d_i + e_i \\
  \text{Long:}  \quad y_i &= \beta_0 + \beta_1 d_i + \beta_2 x_i + u_i
\end{align*}

\begin{enumerate}
  \item Write the omitted variable bias formula relating $\gamma_1$ to $\beta_1$.

  \textbf{Answer.} $\gamma_1 = \beta_1 + \tilde\delta\, \beta_2$, where $\tilde\delta$ is the slope coefficient from regressing $x_i$ on $d_i$ (and a constant).

  \item Since treatment is randomized, $d_i$ is independent of $x_i$.  Does omitted variable bias disappear?

  \textbf{Answer.} Yes. Under randomization $\tilde\delta = 0$ in population, so $\gamma_1 = \beta_1$.  There is no omitted variable bias.

  \item Even without bias, the researcher might still want to include $x_i$.  Why?

  \textbf{Answer.} Including a relevant covariate $x_i$ absorbs residual variance: $\text{Var}(u_i) < \text{Var}(e_i)$ whenever $\beta_2 \neq 0$. Since the variance of $\hat\beta_1$ is $\sigma_u^2 / \sum(d_i - \bar d)^2$ (up to a leverage adjustment that is negligible under randomization), a smaller residual variance yields a smaller standard error. One gets a more precise treatment-effect estimate at no cost in bias.
\end{enumerate}

%%% PROBLEM 4 %%%
\item Suppose $y_i = \bm{x}_i'\bfbeta + \epsilon_i$ with $\mathbb{E}[\epsilon_i \mid \bm{x}_i] = 0$ and $\text{Var}(\epsilon_i \mid \bm{x}_i) = \sigma_i^2$ (which may vary across observations).

\begin{enumerate}
  \item Write the variance of the OLS estimator $\hat{\bfbeta}$ conditional on $\bm{X}$ (the ``sandwich'' formula).

  \textbf{Answer.}
  $\text{Var}(\hat{\bfbeta} \mid \bm{X}) = (\bm{X}'\bm{X})^{-1}\!\left(\displaystyle\sum_{i=1}^n \sigma_i^2\,\bm{x}_i\bm{x}_i'\right)\!(\bm{X}'\bm{X})^{-1} = (\bm{X}'\bm{X})^{-1}\bm{X}'\bm{D}\bm{X}\,(\bm{X}'\bm{X})^{-1}$,
  where $\bm{D} = \text{diag}(\sigma_1^2, \ldots, \sigma_n^2)$.

  \item Show that under homoskedasticity ($\sigma_i^2 = \sigma^2$ for all $i$), the sandwich simplifies to the classical formula $\sigma^2(\bm{X}'\bm{X})^{-1}$.

  \textbf{Answer.} When $\bm{D} = \sigma^2\bm{I}$:
  $(\bm{X}'\bm{X})^{-1}\bm{X}'\sigma^2\bm{I}\,\bm{X}\,(\bm{X}'\bm{X})^{-1} = \sigma^2(\bm{X}'\bm{X})^{-1}\bm{X}'\bm{X}\,(\bm{X}'\bm{X})^{-1} = \sigma^2(\bm{X}'\bm{X})^{-1}$.

  \item When $\bm{D}$ is known and $\text{Var}(\bfeps \mid \bm{X}) = \bm{\Sigma}$ is a general positive definite matrix, OLS is no longer the best linear unbiased estimator.  Write the formula for the estimator that \emph{is} BLUE in this setting.

  \textbf{Answer.} The Generalized Least Squares (GLS) estimator:
  $\hat{\bfbeta}_{\text{GLS}} = (\bm{X}'\bm{\Sigma}^{-1}\bm{X})^{-1}\bm{X}'\bm{\Sigma}^{-1}\bm{y}$.
  GLS is equivalent to OLS on the transformed model $\bm{\Sigma}^{-1/2}\bm{y} = \bm{\Sigma}^{-1/2}\bm{X}\bfbeta + \bm{\Sigma}^{-1/2}\bfeps$, where the transformed errors are now spherical.
\end{enumerate}

%%% PROBLEM 5 %%%
\item Suppose your data matrix is $(\bm{y},\bm{X})$ and you construct a new regressor matrix $\bm{Z} = \bm{X}\bm{C}$ where $\bm{C}$ is an invertible matrix.  For example, $\bm{X} = [\bm{X}_1, \bm{X}_2]$ and $\bm{Z} = [\bm{X}_1, \bm{X}_2 - \bm{X}_1]$.  Show that the residuals from regressing $\bm{y}$ on $\bm{X}$ and regressing $\bm{y}$ on $\bm{Z}$ are identical, and therefore $\hat\sigma^2$ is the same in both regressions.

\textbf{Answer.} Since $\bm{C}$ is invertible, $\text{col}(\bm{Z}) = \text{col}(\bm{XC}) = \text{col}(\bm{X})$: any $\bm{Za}$ equals $\bm{X}(\bm{Ca})$, and any $\bm{Xb}$ equals $\bm{Z}(\bm{C}^{-1}\bm{b})$.  The two column spaces are the same, so the projection matrices are the same: $\bm{P}_Z = \bm{P}_X$.  Therefore $\hat{\bm{e}}_Z = (\bm{I} - \bm{P}_Z)\bm{y} = (\bm{I} - \bm{P}_X)\bm{y} = \hat{\bm{e}}_X$, and $\hat\sigma^2_Z = \hat{\bm{e}}_Z'\hat{\bm{e}}_Z/(n-k) = \hat{\bm{e}}_X'\hat{\bm{e}}_X/(n-k) = \hat\sigma^2_X$.

Note: the individual coefficients \emph{do} change ($\hat{\bm{\gamma}} = \bm{C}^{-1}\hat{\bfbeta}$), but fitted values, residuals, and all goodness-of-fit measures are invariant to invertible reparameterizations.

%%% PROBLEM 6 (EXTRA CREDIT) %%%
\item \textbf{EXTRA CREDIT.}  Consider the model $y_i = \beta x_i + \epsilon_i$ (no intercept) with $\mathbb{E}[\epsilon_i \mid x_i] = 0$ and $\text{Var}(\epsilon_i \mid x_i) = \sigma^2$ for all $i$.  The OLS estimator is $\hat\beta = \sum_{i=1}^n x_i y_i \big/ \sum_{i=1}^n x_i^2$.  A student proposes the ``ratio estimator''
$$
  \tilde\beta = \frac{1}{n}\sum_{i=1}^n \frac{y_i}{x_i}
  \qquad (x_i \neq 0 \text{ for all } i).
$$

\begin{enumerate}
  \item Show that $\tilde\beta$ is unbiased for $\beta$.

  \textbf{Answer.} $\mathbb{E}[\tilde\beta \mid \bm{X}] = \frac{1}{n}\sum_{i=1}^n \mathbb{E}[y_i/x_i \mid \bm{X}] = \frac{1}{n}\sum_{i=1}^n \frac{\beta x_i}{x_i} = \frac{1}{n}\sum_{i=1}^n \beta = \beta$.

  \item Show that $\text{Var}(\tilde\beta \mid \bm{X}) \geq \text{Var}(\hat\beta \mid \bm{X})$.

  \emph{Hint:} the Cauchy--Schwarz inequality gives $\bigl(\sum a_i b_i\bigr)^2 \leq \bigl(\sum a_i^2\bigr)\bigl(\sum b_i^2\bigr)$.

  \textbf{Answer.}
  $\text{Var}(\tilde\beta \mid \bm{X}) = \frac{1}{n^2}\sum \text{Var}(y_i/x_i \mid \bm{X}) = \frac{\sigma^2}{n^2}\sum \frac{1}{x_i^2}$.
  $\text{Var}(\hat\beta \mid \bm{X}) = \sigma^2 \big/ \sum x_i^2$.

  We need $\frac{1}{n^2}\sum \frac{1}{x_i^2} \geq \frac{1}{\sum x_i^2}$, i.e., $\bigl(\sum x_i^2\bigr)\bigl(\sum 1/x_i^2\bigr) \geq n^2$.  Apply Cauchy--Schwarz with $a_i = |x_i|$ and $b_i = 1/|x_i|$:
  $$\Bigl(\sum_{i=1}^n |x_i|\cdot\frac{1}{|x_i|}\Bigr)^{\!2} = n^2 \;\leq\; \Bigl(\sum_{i=1}^n x_i^2\Bigr)\!\Bigl(\sum_{i=1}^n \frac{1}{x_i^2}\Bigr).$$
  This is exactly the Gauss--Markov theorem in action: OLS minimizes variance among all linear unbiased estimators, and $\tilde\beta$ is linear and unbiased but not OLS.

  \item Under what condition on the $x_i$ values are the two variances equal?

  \textbf{Answer.} Cauchy--Schwarz holds with equality when $a_i/b_i$ is constant, i.e., $x_i^2/(1/x_i^2) = x_i^4$ is constant across all $i$.  This means all $|x_i|$ must be equal.  When every observation has the same $|x_i|$, the ratio estimator and OLS coincide.
\end{enumerate}

\end{enumerate}
