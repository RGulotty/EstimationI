\item \textbf{Log-Likelihood of the Normal.}\\
The log-density of a $N(\mu, \sigma^2)$ random variable $x$ is:
$$\log f(x|\mu,\sigma^2) = -\frac{1}{2}\log(2\pi) - \frac{1}{2}\log(\sigma^2) - \frac{(x-\mu)^2}{2\sigma^2}$$
\begin{enumerate}[label=(\alph*)]
\item Treating $\sigma^2$ as known, compute $\dfrac{\partial}{\partial \mu}\log f(x|\mu,\sigma^2)$.
\vspace{0.7in}
\item For $n$ independent observations, the log-likelihood is $\ell(\mu) = \sum_{i=1}^n \log f(x_i|\mu,\sigma^2)$.  Set $\dfrac{d\ell}{d\mu} = 0$ and solve for $\hat\mu$.
\vspace{0.8in}
\end{enumerate}

\item
\begin{enumerate}[label=(\alph*)]
\item $3p^2(1-p)^2 - 2p^3(1-p) = p^2(1-p)[3(1-p)-2p] = p^2(1-p)(3-5p)$
\item $p^2(1-p)(3-5p) = 0$. Solutions: $p=0$, $p=1$, $p=3/5$. The interior maximum is $p=3/5$.
\end{enumerate}

\item
\begin{enumerate}[label=(\alph*)]
\item $-3/\lambda$
\item $5/\lambda^2$
\item $-3/\lambda + 5/\lambda^2 = 0 \implies 5/\lambda^2 = 3/\lambda \implies \lambda = 5/3$.
\end{enumerate}

\item
\begin{enumerate}[label=(\alph*)]
\item $\dfrac{\partial}{\partial\mu}\log f = \dfrac{x-\mu}{\sigma^2}$
\item $\dfrac{d\ell}{d\mu} = \dfrac{1}{\sigma^2}\sum_{i=1}^n(x_i - \mu) = 0 \implies n\mu = \sum x_i \implies \hat\mu = \bar{x} = \dfrac{1}{n}\sum x_i$.
\end{enumerate}

\item

\item At $\hat\lambda = 5/3$: $\ell''(5/3) = 3/(25/9) - 10/(125/27) = 27/25 - 270/125 = 27/25 - 54/25 = -27/25 < 0$. Confirmed: maximum.
